// column-customization.component.ts import { Component } from '@angular/core';

@Component({ selector: 'app-column-customization', templateUrl: './column-customization.component.html', styleUrls: ['./column-customization.component.scss'] }) export class ColumnCustomizationComponent { columns = [ { name: 'Proposal name', locked: true, visible: true }, { name: 'CRM Opp. Number', locked: false, visible: true }, { name: 'Status', locked: false, visible: true }, { name: 'Client', locked: false, visible: true }, { name: 'GCI Number', locked: false, visible: true }, { name: 'Date Received', locked: false, visible: true }, { name: 'Due Date', locked: false, visible: true }, { name: 'Owning Team', locked: false, visible: true }, { name: 'Owning Branch', locked: false, visible: true }, { name: 'Bid Owner', locked: false, visible: true }, { name: 'Pricing Bid Owner', locked: false, visible: true }, { name: 'Geo Sponsor', locked: false, visible: true }, { name: 'Completed Date', locked: false, visible: true } ];

maxLocked = 4;

get lockedCount() { return this.columns.filter(c => c.locked).length; }

toggleLock(index: number) { if (!this.columns[index].locked && this.lockedCount >= this.maxLocked) return; this.columns[index].locked = !this.columns[index].locked; this.sortColumns(); }

toggleVisibility(index: number) { this.columns[index].visible = !this.columns[index].visible; }

sortColumns() { this.columns.sort((a, b) => (b.locked ? 1 : 0) - (a.locked ? 1 : 0)); }

move(index: number, direction: number) { const newIndex = index + direction; if (newIndex < 0 || newIndex >= this.columns.length) return; [this.columns[index], this.columns[newIndex]] = [this.columns[newIndex], this.columns[index]]; } }

// column-customization.component.html

<div class="column-customization-container">
  <mat-card *ngFor="let column of columns; let i = index" class="column-item">
    <div class="item-row">
      <button mat-icon-button (click)="move(i, -1)">
        <mat-icon>arrow_upward</mat-icon>
      </button>
      <button mat-icon-button (click)="move(i, 1)">
        <mat-icon>arrow_downward</mat-icon>
      </button>
      <mat-checkbox [(ngModel)]="column.visible">
        {{ column.name }}
      </mat-checkbox>
      <button mat-icon-button (click)="toggleLock(i)">
        <mat-icon>{{ column.locked ? 'lock' : 'lock_open' }}</mat-icon>
      </button>
    </div>
  </mat-card>
  <mat-divider *ngIf="lockedCount > 0"></mat-divider>
</div>// column-customization.component.scss .column-customization-container { width: 400px; margin: auto; }
